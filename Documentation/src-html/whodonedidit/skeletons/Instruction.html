001package whodonedidit.skeletons;
002
003/**
004 * Interface Instruction facilitates communication from the Game
005 * Manager to the Player. The Game Manager sends instructions to a
006 * Player for them to carry out. Each subclass must create a
007 * Constructor that will have the needed information to do the
008 * instruction like Cards or description of what to do.
009 *
010 * @author austinrobarts
011 */
012public interface Instruction
013{
014
015    /**
016     * Method enforced in all subclasses so any Instruction can be
017     * executed out by the Player
018     *
019     * @param player the player that will be doing the instruction
020     */
021    public void doInstruction(Player player);
022}
