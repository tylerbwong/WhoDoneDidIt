001package whodonedidit.skeletons;
002
003/**
004 * Interface Action will be implemented by subclasses to represent all
005 * possible Actions a player can take during a game of Who Done Did It
006 *
007 * All Actions will be constructed by players and then sent to the
008 * Game Manager to be executed. Some example actions are playing a
009 * suggestion card or switching a destination marker.
010 *
011 * @author austinrobarts
012 */
013public interface Action
014{
015
016    /**
017     * This method will be implemented by all Action subclasses to
018     * conduct a certain action. The doAction method will be called by
019     * the game manager's executePlayerAction method
020     *
021     * @param manager A reference to the game manager
022     */
023    public void doAction(GameManager manager);
024}
